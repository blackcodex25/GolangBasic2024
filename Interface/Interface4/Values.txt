ค่าต่างๆ
ถ้าค่าที่เป็นรูปธรรมภาพในอินเทอร์เฟซเป็น nil เมธอดจะถูก
เรียกใช้ด้วยตัวรับที่เป็น nil
ในบางภาษา สิ่งนี้จะทำให้เกิดข้อยกเว้น null pointer แต่ใน
ภาษา Go มักจะเขียนเมธอดที่จัดการกับการถูกเรียกใช้ด้วยตัวรับ
ที่เป็น nil ได้อย่างดี (เช่นกับเมธอด M ในตัวอย่างนี้)
โปรดทราบว่า อินเทอร์เฟศที่ถือค่า nil ที่เป็นรูปธรรมจะไม่เป็น
nil เอง

/*
	ประกาศอินเตอร์เฟซชื่อ I ประกาศเมธอด M

ซึ่งไม่มีพารามิเตอร์และไม่มีค่าที่คืนกลับ
*/
type I interface {
	M()
}

// ประกาศสตรัคชื่อตัวแปร T กำหนดฟิลด์ชื่อ S ชนิดข้อมูล string
type T struct {
	S string
}

// ประกาศพารามิเตอร์ตัวรับชื่อตัวแปร t ชนิดข้อมูลของสตรัค T
// ประกาศเมธอด M() สำหรับ Pointer ตัวรับ t ไปยังสตรัค T (*T)
func (t *T) M() {
	// ตรวจสอบว่าตัวรับ t เท่ากับ nil หรือไม่
	if t == nil {
		// ถ้าใช่ พิมพ์ผลลัพธ์ออกจอ
		f.Println("<nil>")
		// หยุดการทำงานของเมธอดด้วยการ return เพื่อไม่ให้โปรแกรมพยายามเข้าถึง
		// ฟิลด์ S ของ t ซึ่งจะทำให้เกิดข้อผิดพลาด (nil pointer dereference)
		return // ประกาศฟังก์ชัน return
	}
	// ถ้าไม่ใช่ nil พิมพ์ค่าของฟิลด์ S ของตัวรับ t ออกจอภาพ
	f.Println(t.S) /* ประกาศฟังก์ชัน Println() เรียกใช้ตัวรับ t
	และเข้าถึงฟิลด์ S ของสตรัค T พิมพ์ค่าของฟิลด์ S ของตัวรับ t ออกจอ */
}

// ประกาศฟังก์ชัน describe กำหนดพารามิเตอร์ตัวรับ i ชนิดข้อมูลของอินเตอร์เฟซ I
func describe(i I) {
	f.Printf("(%v, %T)\n", i, i) //ประกาศฟังก์ชัน Printf() ไม่ขึ้นบรรทัดใหม่
	// พิมพ์ค่าตัวแปร i และชนิดข้อมูลของ i ออกจอภาพ
}

var i I  // ประกาศตัวแปร i ชนิดข้อมูลของอินเตอร์เฟซ I
var t *T // ประกาศตัวแปร t ชนิดชี้ข้อมูลไปยังสตรัคชื่อ T และกำหนดค่าให้เป็น nil

func main() {
	// กำหนดค่า t ให้กับ i (i = t)
	i = t
	describe(i) // เรียกใช้ฟังก์ชัน describe กำหนดค่า i พิมพ์ผลลัพ์ออกจอภาพ
	i.M()       // เรียกใช้เมธอด M() ของตัวแปร i ชนิดข้อมูลของอินเตอร์เฟซ I

	// กำหนดค่า i ชนิดชี้ไปยังข้อมูลสตรัค T กำหนดอากิวเมนต์ S เป็น Hi Dev Go
	i = &T{"hello"}
	// เรียกใช้ฟังก์ชัน describe กำหนดค่า i ซึ่งมีชนิดข้อมูลสตรัค T
	describe(i) // พิมพ์ผลลัพธ์ออกจอ
	// เรียกใช้เมธอด M() ของตัวแปร i ชนิดข้อมูลของอินเตอร์เฟซ I
	i.M()
	// กำหนดค่า i เท่ากับ t ซึ่ง i มีชนิดข้อมูลของอินเตอร์เฟซ I
	// ให้ชี้ไปยังตัวรับพอยน์เตอร์ตัวแปร t ซึ่งมีชนิดข้อมูลสตรัค T (*T) ที่มีค่า nil
}
