func Walking(t *tree.Tree, ch chan int) {
	/* เรียกใช้ฟังก์ชัน Walk()
	จะเดินผ่านต้นไม้ทวิภาคโดยใช้รูปแบบการเดินผ่านแบบ
	เรียงลำดับ (in-order traversal)
	ค่าของโหนดที่เจอระหว่างการเดินผ่านจะถูกส่งไปยัง
	channels พารามิเตอร์ ch ทีละค่า
	*/
	Walk(t, ch)
	/* การใช้ defer เพื่อปิด Channels
	คำสั่ง defer ใช้เพื่อให้แน่ใจว่า channels พารามิเตอร์ ch
	ปิดลงหลังจากการเดินผ่านต้นไม้เสร็จสิ้น ซึ่งหมายความว่าจะ
	ไม่มีการส่งข้อมูลใดๆ ผ่าน channels นี้อีก
	*/
	defer close(ch)
}
